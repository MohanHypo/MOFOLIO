function severityBanner(income,expense){
  if(!income&&!expense)return null;
  const ratio=income===0?999:expense/income;
  let level,bg,border,tc,icon,msg,sugg;
  if(ratio<=0.5){level="Excellent";bg="#E1F5EE";border="#1D9E75";tc="#085041";icon="✓";msg="Saving 50%+ of income. Outstanding!";sugg=["Increase SIP contributions.","Explore NPS for extra tax benefits.","Consider direct equity for long-term growth."];}
  else if(ratio<=0.7){level="Healthy";bg="#E6F1FB";border="#378ADD";tc="#0C447C";icon="↑";msg="Good savings buffer. Keep it up!";sugg=["Build 6-month emergency fund.","Do 10% annual step-up on SIPs.","Review subscriptions monthly."];}
  else if(ratio<=0.9){level="Caution";bg="#FAEEDA";border="#BA7517";tc="#412402";icon="!";msg="Savings are thin. Review your budget.";sugg=["Cap top 2 expense categories.","Avoid new EMIs.","Explore side income."];}
  else if(ratio<=1){level="Warning";bg="#FAECE7";border="#D85A30";tc="#4A1B0C";icon="⚠";msg="Almost nothing is being saved!";sugg=["Apply 50-30-20 rule.","Pause non-essential spending.","Start ₹500/month SIP today."];}
  else{level="Critical";bg="#FCEBEB";border="#A32D2D";tc="#501313";icon="✕";msg=`Deficit of ${fmt(expense-income)}! Expenses exceed income.`;sugg=["Stop discretionary spending NOW.","Clear highest-interest debt first.","Seek additional income urgently."];}
  const d=el("div",{style:{background:bg,border:`1.5px solid ${border}`,borderRadius:"12px",padding:"1rem",marginBottom:"1rem"}});
  const top=el("div",{style:{display:"flex",alignItems:"center",gap:"8px",marginBottom:"6px"}});
  top.appendChild(el("span",{style:{width:"26px",height:"26px",borderRadius:"50%",background:border,color:"#fff",display:"flex",alignItems:"center",justifyContent:"center",fontSize:"13px",fontWeight:"700",flexShrink:"0"}},icon));
  top.appendChild(el("span",{style:{fontWeight:"500",fontSize:"14px",color:tc}},`Financial Health: ${level}`));
  top.appendChild(el("span",{style:{marginLeft:"auto",fontSize:"12px",color:tc,fontWeight:"500"}},`${Math.round(ratio*100)}%`));

function buildCorpus(wrap){
  let mode="build",monthly=10000,target=10000000,rate=12,years=20,inflation=6;
  const resWrap=el("div",{});
  function calc(){
    const r=rate/100/12,n=years*12;
    let fv,inv,needed,realFv;
    if(mode==="build"){fv=monthly*((Math.pow(1+r,n)-1)/r)*(1+r);inv=monthly*n;realFv=fv/Math.pow(1+inflation/100,years);}
    else{needed=target/((((Math.pow(1+r,n)-1)/r)*(1+r)));inv=needed*n;fv=target;realFv=target/Math.pow(1+inflation/100,years);}
    const rows=[];const base=mode==="build"?monthly:needed;
    for(let y=1;y<=years;y++){const ny=y*12;const nf=base*((Math.pow(1+r,ny)-1)/r)*(1+r);const i2=base*ny;rows.push({y,inv:i2,nom:nf,real:nf/Math.pow(1+inflation/100,y),gain:nf-i2});}
